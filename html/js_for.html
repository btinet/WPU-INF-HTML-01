Anwendungsbeispiel "for-Schleife"

Code für obiges Beispiel

                
            (function () {

    // Container definieren
    let $forContainer = document.querySelector('#for-schleife-example');
    // Variable für den Hintergrund/Schriftfarbe initiieren
    let $bg = '';
    // x Durchläufe
    for(let $i=1; $i<=12;$i++)
    {
        // Wenn $i ist gerade
        if($i%2)
        {
            // Heller Hintergrund
            $bg = 'bg-light';
        }
        // Ansonsten (falls $i ist ungerade)
        else
        {
            // Dunkler Hintergrund, helle Schriftfarbe
            $bg = 'bg-dark text-light';
        }
        // Variable mit Elementinhalt initiieren,
        // Hintergrund/Schriftfarbe per CSS angeben,
        // Durchlaufnummer eintragen
        let $html = '<div class="'+$bg+'">Durchlauf '+$i+'</div>';
        // Funktion write() aufrufen
        write($forContainer,$html);
    }

    // Funktion zum Schreiben generierter Elemente in einen zuvor definierten Container
    function write($container, $element)
    {
        // Element dem Container hinzufügen
        $container.innerHTML += $element;
    }

})()
